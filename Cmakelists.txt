cmake_minimum_required(VERSION 3.16)
project(CinemaOrder LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 自动启用 MOC 处理 QObject/Q_SIGNALS/Q_SLOTS
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

option(QT_ROOT "Qt 安装根路径（例如 /mingw64）" "")

# 若未提供，尝试自动检测常见的 MSYS2/Mingw-w64 Qt 安装路径
if(NOT QT_ROOT)
	if(DEFINED ENV{MINGW_PREFIX})
		set(QT_ROOT $ENV{MINGW_PREFIX})
	elseif(EXISTS "/mingw64")
		set(QT_ROOT "/mingw64")
	elseif(EXISTS "C:/msys64/mingw64")
		set(QT_ROOT "C:/msys64/mingw64")
	endif()
endif()

if(QT_ROOT)
	list(APPEND CMAKE_PREFIX_PATH 
		"${QT_ROOT}/lib/cmake/Qt6" 
		"${QT_ROOT}/lib/cmake"
	)
	message(STATUS "Using QT_ROOT=${QT_ROOT}")
endif()

find_package(Qt6 COMPONENTS Core Sql Widgets QUIET)
if(NOT Qt6_FOUND)
	# 最后尝试：使用显式的常见 MSYS2 路径
	if(EXISTS "/mingw64/lib/cmake/Qt6")
		list(APPEND CMAKE_PREFIX_PATH "/mingw64/lib/cmake/Qt6")
	endif()
	find_package(Qt6 REQUIRED COMPONENTS Core Sql Widgets)
endif()

file(GLOB_RECURSE SRC_FILES 
	src/*.cpp
	src/*/*.cpp
)

add_executable(cinema_order ${SRC_FILES})
target_include_directories(cinema_order PRIVATE 
	src
	src/view
	src/model
	src/controller
	src/auth
)

target_link_libraries(cinema_order 
	Qt6::Core 
	Qt6::Sql 
	Qt6::Widgets
)

# ========== 资源处理部分 ==========
set(RESOURCES_DIR "${CMAKE_SOURCE_DIR}/resources")
message(STATUS "资源目录：${RESOURCES_DIR}")

# 检查并添加资源文件
if(EXISTS "${RESOURCES_DIR}/resources.qrc")
	message(STATUS "找到 resources.qrc")
	
	# 添加资源编译
	qt6_add_resources(RESOURCES_RCC "${RESOURCES_DIR}/resources.qrc")
	
	# 添加资源文件到目标
	target_sources(cinema_order PRIVATE ${RESOURCES_RCC})
	
	# 输出生成的资源文件名
	foreach(RCC ${RESOURCES_RCC})
		message(STATUS "生成资源文件：${RCC}")
	endforeach()
else()
	message(WARNING "在 ${RESOURCES_DIR}/resources.qrc 未找到 resources.qrc")
	# 若没有资源文件，可以创建一个空的
	# file(WRITE "${RESOURCES_DIR}/empty.qrc" "<RCC/>")
	# qt6_add_resources(RESOURCES_RCC "${RESOURCES_DIR}/empty.qrc")
endif()

# 确保 MOC 处理
set(MOC_HEADERS
	${CMAKE_SOURCE_DIR}/src/view/ticketformdialog.h
	${CMAKE_SOURCE_DIR}/src/view/seatselectiondialog.h
	${CMAKE_SOURCE_DIR}/src/view/mainwindow.h
	${CMAKE_SOURCE_DIR}/src/view/usermainwindow.h
	${CMAKE_SOURCE_DIR}/src/view/logindialog.h
	${CMAKE_SOURCE_DIR}/src/view/registerdialog.h
	${CMAKE_SOURCE_DIR}/src/view/userinfodialog.h
)
qt6_wrap_cpp(MOC_SOURCES ${MOC_HEADERS})
if(MOC_SOURCES)
	target_sources(cinema_order PRIVATE ${MOC_SOURCES})
endif()

message(STATUS "Qt6 目录=${Qt6_DIR}")
message(STATUS "CMake 前缀路径=${CMAKE_PREFIX_PATH}")
